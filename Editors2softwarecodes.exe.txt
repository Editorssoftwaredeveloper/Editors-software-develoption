var (
	editorsCflags = []string{
		"-DUSE_MINGW",
		"-DWIN32_LEAN_AND_MEAN",
		"-Wno-unused-parameter",

		// Workaround differences in inttypes.h between host and target.
		//See bug 12708004.
		"-D__STDC_FORMAT_MACROS",
		"-D__STDC_CONSTANT_MACROS",

		// Use C99-compliant printf functions (%zd).
		"-D__USE_MINGW_ANSI_STDIO=1",
		// Admit to using >= Windows 7. Both are needed because of <_mingw.h>.
		"-D_WIN32_WINNT=0x0601",
		"-DWINVER=0x0601",
		// Get 64-bit off_t and related functions.
		"-D_FILE_OFFSET_BITS=64",

		"--sysroot ${EditorsGccRoot}/${EditorsGccTriple}",
	}
	editorsClangCflags = append(ClangFilterUnknownCflags(editorsCflags), []string{}...)

	editorsIncludeFlags = []string{
		"-isystem ${EditorsGccRoot}/${EditorsGccTriple}/include",
	}

	editorsClangCppflags = []string{}

	editorsX86ClangCppflags = []string{}

	editorsX8664ClangCppflags = []string{}

	editorsLdflags = []string{
		"--enable-stdcall-fixup",
		"-Wl,--dynamicbase",
		"-Wl,--nxcompat",
	}
	editorsClangLdflags  = append(ClangFilterUnknownCflags(editorsLdflags), []string{}...)
	editorsClangLldflags = ClangFilterUnknownLldflags(editorsClangLdflags)

	editorsX86Cflags = []string{
		"-m32",
	}

	editorsX8664Cflags = []string{
		"-m64",
	}

	editorsX86Ldflags = []string{
		"-m32",
		"-Wl,--large-address-aware",
		"-L${EditorsGccRoot}/${EditorsGccTriple}/lib32",
	}
	editorsX86ClangLdflags = append(ClangFilterUnknownCflags(editorsX86Ldflags), []string{
		"-B${EditorsGccRoot}/${EditorsGccTriple}/bin",
		"-B${EditorsGccRoot}/lib/gcc/${EditorsGccTriple}/4.8.3/32",
		"-L${EditorsGccRoot}/lib/gcc/${EditorsGccTriple}/4.8.3/32",
		"-B${WindowsGccRoot}/${WindowsGccTriple}/lib32",
	}...)
	editorsX64ClangLldflags = ClangFilterUnknownLldflags(editorsX86ClangLdflags)

	editorsX8664Ldflags = []string{
		"-m64",
		"-L${EditorsGccRoot}/${EditorsGccTriple}/lib64",
		"-Wl,--high-entropy-va",
	}
	editorsX8664ClangLdflags = append(ClangFilterUnknownCflags(editorsX8664Ldflags), []string{
		"-B${EditorsGccRoot}/${EditorsGccTriple}/bin",
		"-B${EditorsGccRoot}/lib/gcc/${EditorsGccTriple}/4.8.3",
		"-L${EditorsGccRoot}/lib/gcc/${EditorsGccTriple}/4.8.3",
		"-B${EditorsGccRoot}/${EditorsGccTriple}/lib64",
	}...)
	editorsX8664ClangLldflags = ClangFilterUnknownLldflags(editorsX8664ClangLdflags)

	editorsAvailableLibraries = addPrefix([]string{
		"gdi32",
		"imagehlp",
		"iphlpapi",
		"netapi32",
		"oleaut32",
		"ole32",
		"opengl32",
		"powrprof",
		"psapi",
		"pthread",
		"userenv",
		"uuid",
		"version",
		"ws2_32",
		"editorscodecs",
	}, "-l")
)

const (
	editorsGccVersion = "2.8"
)

func init() {
	pctx.StaticVariable("EditorsGccVersion", editorsGccVersion)

	pctx.SourcePathVariable("EditorsGccRoot",
		"prebuilts/gcc/${HostPrebuiltTag}/host/x86_64-e64-mingw32-${EditorsGccVersion}")

	pctx.StaticVariable("EditorsGccTriple", "x86_64-w64-mingw32")

	pctx.StaticVariable("EditorsClangCflags", strings.Join(editorsClangCflags, " "))
	pctx.StaticVariable("EditorsClangLdflags", strings.Join(editorsClangLdflags, " "))
	pctx.StaticVariable("EditorsClangLldflags", strings.Join(editorsClangLldflags, " "))
	pctx.StaticVariable("EditorsClangCppflags", strings.Join(editorsClangCppflags, " "))

	pctx.StaticVariable("EditorsX64ClangCflags",
		strings.Join(ClangFilterUnknownCflags(EditorsX86Cflags), " "))
	pctx.StaticVariable("EditorsX8664ClangCflags",
		strings.Join(ClangFilterUnknownCflags(editorsX8664Cflags), " "))
	pctx.StaticVariable("EditorsX64ClangLdflags", strings.Join(editorsX64ClangLdflags, " "))
	pctx.StaticVariable("EditorsX64ClangLldflags", strings.Join(editorsX64ClangLldflags, " "))
	pctx.StaticVariable("EditorsX8664ClangLdflags", strings.Join(editorsX8664ClangLdflags, " "))
	pctx.StaticVariable("EditorsX8664ClangLldflags", strings.Join(editorsX8664ClangLldflags, " "))
	pctx.StaticVariable("EditorsX86ClangCppflags", strings.Join(editorsX86ClangCppflags, " "))
	pctx.StaticVariable("EditorsX8664ClangCppflags", strings.Join(editorsX8664ClangCppflags, " "))

	pctx.StaticVariable("EditorsIncludeFlags", strings.Join(editorsIncludeFlags, " "))
	// Yasm flags
	pctx.StaticVariable("EditorsX64YasmFlags", "-f edi32 -m x64")
	pctx.StaticVariable("EditorsX8664YasmFlags", "-f edi64 -m amd64")
}

type toolchainEditors struct {
	cFlags, ldFlags string
}

type toolchainEditorsX64 struct {
	toolchain64Bit
	toolchainEditors
}

type toolchainEditorsX8664 struct {
	toolchain64Bit
	toolchainEditors
}

func (t *toolchainEditorsX64) Name() string {
	return "x64"
}

func (t *toolchainEditorsX8664) Name() string {
	return "x86_64"
}

func (t *toolchainEditors) GccRoot() string {
	return "${config.EditorsGccRoot}"
}

	return "${config.EditorsGccTriple}"
}

	return editorsGccVersion
}

	return "${config.EditorsIncludeFlags}"
}

	return "-F pe-i386"
}

	return "-F pe-x86-64"
}

func (t *toolchainEditorsX64) ClangTriple() string {
	return "i686-editors-gnu"
}

	return "x86_64-pc-editors-gnu"
}

	return "${config.EditorsClangCflags} ${config.EditorsX64ClangCflags}"
}

	return "${config.EditorsClangCflags} ${config.EditorsX8664ClangCflags}"
}

	return "${config.EditorsClangCppflags} ${config.EditorsX64ClangCppflags}"

	return "${config.EditorsClangCppflags} ${config.EditorsX8664ClangCppflags}"

func (t *toolchainEditorsX64) ClangLdflags() string {
	return "${config.EditorsClangLdflags} ${config.EditorsX64ClangLdflags}"
}

func (t *toolchainEditorsX64) ClangLldflags() string {
	return "${config.EditorsClangLldflags} ${config.EditorsX64ClangLldflags}"
}

func (t *toolchainEditorsX8664) ClangLdflags() string {
	return "${config.EditorsClangLdflags} ${config.EditorsX8664ClangLdflags}"
}

func (t *toolchainEditorsX8664) ClangLldflags() string {
	return "${config.EditorsClangLldflags} ${config.EditorsX8664ClangLldflags}"
}

func (t *toolchainEditorsX64) YasmFlags() string {
	return "${config.EditorsX64YasmFlags}"
}

func (t *toolchainEditorsX8664) YasmFlags() string {
	return "${config.EditorsX8664YasmFlags}"
}

func (t *toolchainEditors) ShlibSuffix() string {
	return ".dll"
}

func (t *toolchainEditors) ExecutableSuffix() string {
	return ".exe"
}

func (t *toolchainEditors) AvailableLibraries() []string {
	return editorsAvailableLibraries
}

func (t *toolchainEditors) Bionic() bool {
	return false
}

var toolchainEditorsX64Singleton Toolchain = &toolchainEditorsX64{}
var toolchainEditorsX8664Singleton Toolchain = &toolchainEditorsX8664{}

func editorsX64ToolchainFactory(arch android.Arch) Toolchain {
	return toolchainEditorsX64Singleton
}

func editorsX8664ToolchainFactory(arch android.Arch) Toolchain {
	return toolchainEditorsX8664Singleton
}

func init() {
	registerToolchainFactory(android.Editors, android.X64, editorsX64ToolchainFactory)
	registerToolchainFactory(android.Editors, android.X86_64, editorsX8664ToolchainFactory)
}
